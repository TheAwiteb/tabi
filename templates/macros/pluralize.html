{% macro pluralize(language, number) %} {%- set key_prefix = "" -%} {# Arabic-specific pluralization rules #} {%- if language == "ar" -%} {%- set modulo = number % 100 -%} {%- if number == 0 -%} {%- set key_prefix = "zero_" -%} {%- elif number == 1 %} {%- set key_prefix = "one_" -%} {%- elif number == 2 %} {%- set key_prefix = "two_" -%} {%- elif number >= 3 and number <= 10 or modulo >= 3 and modulo <= 10 %} {%- set key_prefix = "few_" -%} {%- else -%} {%- set key_prefix = "many_" -%} {%- endif -%} {# Other languages #} {%- else -%} {%- if number == 1 -%} {%- set key_prefix = "one_" -%} {%- else -%} {%- set key_prefix = "many_" -%} {%- endif -%} {%- endif -%} {{-key_prefix-}} {% endmacro %}
